#Ensures Reno will join the NCR instead of trying to and having the focus cancel part of the way through due to declaring a war for its Nevada claims
NEW_join_NCR_before_war = {
	allowed = {
		NOT = { original_tag = NEW }
	}
	enable = {
		is_neighbor_of = NEW
	}
	enable_reverse = {
		NEW = {
			OR = {
				NEW_has_ai_behavior_bishop = yes
				NEW_has_ai_behavior_wright = yes
			}
			NOT = { has_completed_focus = new_stateofreno }
			has_completed_focus = new_nevada
		}
	}
	abort = {
	}
	reversed = yes
	abort_when_not_enabled = yes

	ai_strategy = {
		type = ignore_claim
		id = NEW
		value = 1
	}
}
